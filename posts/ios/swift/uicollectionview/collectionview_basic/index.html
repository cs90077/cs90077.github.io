CollectionView 設計客製化介面

你知道iPhone內建的「照片APP」是用什麼UI製作的嗎？

是利用 UICollectionView 這個UI元件設計出來的。

UICollectionView 可以用來呈現網格多行的樣式，依照水平、垂直不同的顯示方式，可以做出很多不同的變化。

step 1 UI

Storyboard

  • 首先在storyboard設計你的ViewController，將UICollectionView拉進去你的View，並且設定好邊界對應參數。

  • 在這邊有看到CollectionView裡面有包一個CollectionViewCell，這個Cell就是最後畫面會呈現項目，接下來我們先設計出CollectionViewCell。

Cell

  • 建立一個空白的Xib

  • 新增CollectionViewCell到Xib中

  • 設計出你要的畫面（自行排版設計）

  • Class
 1//  PhotoCollectionCell.swift

step 2 Setup

 1// ViewController.swift
 2
 3class ViewController: UIViewController {
 4    
 5    @IBOutlet weak var collectionView: UICollectionView!
 6    
 7    var dataSource: [Int] = []
 8    
 9    lazy var layout: UICollectionViewFlowLayout = {
10        // 建立 UICollectionViewFlowLayout
11        let layout = UICollectionViewFlowLayout()
12
13        // 設置 section 的間距 四個數值分別代表 上、左、下、右 的間距
14        layout.sectionInset = UIEdgeInsets.init(top: 5, left: 5, bottom: 5, right: 5)
15
16        // 設置每一行的間距
17        layout.minimumLineSpacing = 5
18
19        // 設置每個 cell 的尺寸
20        layout.itemSize = CGSize(width: CGFloat(fullScreenSize.width)/3 - 10.0, height: CGFloat(fullScreenSize.width)/3 - 10.0)
21        
22        // 設置 header 及 footer 的尺寸
23        layout.headerReferenceSize = CGSize(
24          width: fullScreenSize.width, height: 40)
25        layout.footerReferenceSize = CGSize(
26          width: fullScreenSize.width, height: 40)
27        return layout
28    }()    
29
30    override func viewDidLoad() {
31        super.viewDidLoad()
32     
33        self.setupCollectionView()
34    }
35    
36    override func viewWillAppear(_ animated: Bool) {
37        super.viewWillAppear(animated)
38        
39        self.reloadData()
40    }
41
42    func setupCollectionView() {
43        collectionView.dataSource = self
44        collectionView.delegate = self
45        
46        // flowlayout
47        collectionView.collectionViewLayout = layout
48        
49        // 註冊自定義的Cell
50        collectionView.register(UINib(nibName: PhotoCollectionCell.identifier, bundle: nil), forCellWithReuseIdentifier: PhotoCollectionCell.identifier)
51        
52        // 註冊自定義的Header
53        collectionView.register(UINib(nibName: PhotoHeaderView.identifier, bundle: nil), forSupplementaryViewOfKind: UICollectionView.elementKindSectionHeader, withReuseIdentifier: PhotoHeaderView.identifier)
54        
55        // 註冊自定義的Footer
56        collectionView.register(UINib(nibName: PhotoFooterView.identifier, bundle: nil), forSupplementaryViewOfKind: UICollectionView.elementKindSectionFooter, withReuseIdentifier: PhotoFooterView.identifier)
57    }
58    
59    func reloadData() {
60        // 在此模擬雙層資料結構
61        for _ in 1...2 {
62            var ary: [Int] = []
63            
64            for i in 1...9 {
65                ary.append(i)
66            }
67            
68            dataSource.append(ary)
69        }
70            
71        // 重整畫面
72        self.collectionView.reloadData()
73    }    
74    
75}

step 3 Delegate

 1// ViewController.swift
 2
 3// MARK: - UICollectionView DataSource
 4
 5extension ViewController: UICollectionViewDelegate, UICollectionViewDataSource {
 6    
 7    // Section數量
 8    func numberOfSections(in collectionView: UICollectionView) -> Int {
 9        return dataSource.count 
10    }
11    
12    // Header View / Footer View
13    func collectionView(_ collectionView: UICollectionView, viewForSupplementaryElementOfKind kind: String, at indexPath: IndexPath) -> UICollectionReusableView {
14        
15        if kind == UICollectionView.elementKindSectionHeader {
16            guard let reusableView = collectionView.dequeueReusableSupplementaryView(ofKind: UICollectionView.elementKindSectionHeader, withReuseIdentifier: PhotoHeaderView.identifier, for: indexPath)
17                    as? PhotoHeaderView else { return UICollectionReusableView() }
18            
19            reusableView.lblTitle.text = "Header"
20            reusableView.lblTitle.textColor = .white
21            reusableView.backgroundColor = .darkGray
22            return reusableView
23        }
24        else if kind == UICollectionView.elementKindSectionFooter {
25            guard let reusableView = collectionView.dequeueReusableSupplementaryView(ofKind: UICollectionView.elementKindSectionFooter, withReuseIdentifier: PhotoFooterView.identifier, for: indexPath)
26                    as? PhotoFooterView else { return UICollectionReusableView() }
27            
28            reusableView.lblTitle.text = "Footer"
29            reusableView.lblTitle.textColor = .white
30            reusableView.backgroundColor = .lightGray
31            return reusableView
32        }
33                
34        return UICollectionReusableView()
35    }
36    
37    // Cell數量
38    func collectionView(_ collectionView: UICollectionView, numberOfItemsInSection section: Int) -> Int {
39        return dataSource[section].count
40    }
41    
42    // Cell View
43    func collectionView(_ collectionView: UICollectionView, cellForItemAt indexPath: IndexPath) -> UICollectionViewCell {
44        
45        guard let cell = collectionView.dequeueReusableCell(withReuseIdentifier: PhotoCollectionCell.identifier, for: indexPath) as? PhotoCollectionCell else {
46            return UICollectionViewCell()
47        }
48        
49        let data = dataSource[indexPath.section][indexPath.row]
50        cell.backgroundColor = .clear
51        cell.imageView.image = UIImage(named: "img_smile")
52        cell.lblTitle.text = data.description
53                
54        return cell
55    }
56    
57    // 點選事件
58    func collectionView(_ collectionView: UICollectionView, didSelectItemAt indexPath: IndexPath) {
59        // click
60        print("index = \(indexPath.row)")
61    }
62    
63}
