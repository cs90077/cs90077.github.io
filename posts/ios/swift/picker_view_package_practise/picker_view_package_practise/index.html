將pickerview做封裝練習，練習製作UI套件

PickerView對於手機APP操作，是一個很方便的項目選擇工具，應用於資料選擇、時間選擇等等。

這邊使用到的技巧

  • UIPickerView
  • Closure

設計

UI

設計好UI，該怎麼綁定與Xib的關係

這邊就不再重新介紹操作了

請參照：客製化AlertView 使用Xib

Custom

 1class LTTPickerView: UIViewController {
 2    
 3    @IBOutlet weak var lblTitle: UILabel!
 4    @IBOutlet weak var btnDone: UIButton!
 5    @IBOutlet weak var btnCancel: UIButton!
 6    @IBOutlet weak var pickerView: UIPickerView!
 7    
 8    // done
 9    typealias CompletionHandler = ((_ row: Int) -> Void)
10    var completionHandler:CompletionHandler?
11    
12    // cnacel
13    typealias CancelHandler = (() -> Void)
14    var cancelHandler:CancelHandler?
15    
16    var pickerTitleText: String?
17    var pickerDoneBtnText: String?
18    var pickerCancelBtnText: String?
19    
20    var dataSource: [String] = []
21    var selectedRow: Int = 0
22    
23    override func viewDidLoad() {
24        super.viewDidLoad()
25        setupView()
26    }
27    
28    @IBAction func donePressed(_ sender: Any) {
29        dismiss(animated: true) {
30            if let handler = self.completionHandler {
31                handler(self.selectedRow)
32            }
33        }
34    }
35    
36    @IBAction func cancelPressed(_ sender: Any) {
37        dismiss(animated: true) {
38            if let handler = self.cancelHandler {
39                handler()
40            }
41        }
42    }
43    
44    private func setupView() {
45        self.lblTitle.text = "title"
46        self.btnDone.setTitle("done", for: .normal)
47        self.btnCancel.setTitle("cancel", for: .normal)
48        
49        self.pickerView.dataSource = self
50        self.pickerView.delegate = self
51        
52        self.pickerView.selectRow(selectedRow, inComponent: 0, animated: true)
53        setupCustomView()
54    }
55    
56    private func setupCustomView() {
57        // setup
58        if let text = self.pickerTitleText {
59            self.lblTitle.text = text
60        }
61        
62        if let text = self.pickerDoneBtnText {
63            self.btnDone.setTitle(text, for: .normal)
64        }
65        
66        if let text = self.pickerCancelBtnText {
67            self.btnCancel.setTitle(text, for: .normal)
68        }
69    }
70    
71    
72}

Func

 1// MARK: Func
 2
 3extension LTTPickerView {
 4    
 5    // 1
 6    func show(_ viewController:UIViewController,
 7              doneHandler:@escaping CompletionHandler,
 8              cancelHandler:@escaping CancelHandler) {
 9        self.modalPresentationStyle = .overFullScreen
10        self.modalTransitionStyle = .crossDissolve
11        self.completionHandler = doneHandler
12        self.cancelHandler = cancelHandler
13        viewController.present(self, animated: true, completion: nil)
14    }
15    
16    // 2
17    func show(_ viewController:UIViewController,
18              dataSource: [String],
19              title: String,
20              doneHandler:@escaping CompletionHandler,
21              cancelHandler:@escaping CancelHandler) {
22        
23        self.dataSource = dataSource
24        self.pickerTitleText = title
25        
26        self.show(viewController, doneHandler: doneHandler, cancelHandler: cancelHandler)
27    }
28    
29}

 2
 3extension LTTPickerView: UIPickerViewDelegate {
 4    
 5    func pickerView(_ pickerView: UIPickerView, titleForRow row: Int, forComponent component: Int) -> String? {
 6
 7        var txt = ""
