格式化

  • dateFormat：格式
  • TimeZone：時區
1let formatter = DateFormatter()
2formatter.dateFormat = "yyyy-MM-dd HH:mm:ss.SSS"
3formatter.locale = Calendar.current.locale
4formatter.timeZone = Calendar.current.timeZone

dateFormat

時間格式有分很多種

// 年-月-日 時:分:秒.毫秒

“yyyy-MM-dd HH:mm:ss.SSS”

ISO DATE

國際標準時間 （有T、Z出現）

“yyyy-MM-ddTHH:mm:ssZ”

2021-12-08T02:59:25Z 轉換成北京時間就是 2021-12-08 10:59:25

可以改變日期顯示的格式內容

1let formatter = DateFormatter()
2formatter.timeZone = Calendar.current.timeZone
3formatter.locale = Calendar.current.locale
4formatter.shortMonthSymbols = ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"]
5formatter.weekdaySymbols = ["(日)","(一)","(二)","(三)","(四)","(五)","(六)"]

透過Calendar取得今天是星期幾

 1///Date輸出
 2func getWeekDayFromDate(input date:Date?) -> String? {
 3    var outString:String = ""
 4    if let date = date {
 5        //let calender:Calendar = Calendar(identifier: Calendar.Identifier.)
 6        let calender = Calendar.current
 7        //calender.timeZone = TimeZone(identifier: "Asia/Taipei")!
 8        
 9        let month = calender.component(.month, from: date)
10        let day = calender.component(.day, from: date)
11        let weekDay = calender.component(.weekday, from: date)
12        let hour = String(format:"%02d", calender.component(.hour, from: date))
13        let minute = String(format:"%02d", calender.component(.minute, from: date))
14        
15        switch weekDay {
16        case 1:
17            outString = "\(month)/\(day) (日) \(hour):\(minute)"
18        case 2:
19            outString = "\(month)/\(day) (一) \(hour):\(minute)"
20        case 3:
21            outString = "\(month)/\(day) (二) \(hour):\(minute)"
22        case 4:
23            outString = "\(month)/\(day) (三) \(hour):\(minute)"
24        case 5:
25            outString = "\(month)/\(day) (四) \(hour):\(minute)"
26        case 6:
27            outString = "\(month)/\(day) (五) \(hour):\(minute)"
28        case 7:
29            outString = "\(month)/\(day) (六) \(hour):\(minute)"
30        default:
31            break
32        }
33        return outString
34    } else {
35        return nil
36    }
37}
