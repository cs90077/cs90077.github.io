使用URL

 1@IBAction func donePressed(_ sender: Any) {
 2    sendSMS()
 3}
 4
 5func sendSMS() {
 6    let phoneNumber = "0911222333"
 7    let body = "我是內容\r\n第二個內容"
 8    
 9    let smsString = "sms:\(phoneNumber)&body=\(body)".addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed) ?? ""
10    if let url = URL(string: smsString) {
11        if #available(iOS 10.0, *) {
12            UIApplication.shared.open(url)
13        } else {
14            UIApplication.shared.openURL(url)
15        }
16    }
17}

使用MFMessageComposeViewController

step 1 import

1import MessageUI

step 2 func

 1@IBAction func donePressed(_ sender: Any) {
 2    showMessageComposer()
 3}
 4
 5func showMessageComposer() {
 6    let phoneNumber = "0911222333"
 7
 8    if (MFMessageComposeViewController.canSendText()) {
 9        let composer = MFMessageComposeViewController()
10        composer.messageComposeDelegate = self
11        composer.recipients = [phoneNumber]
12        composer.body = "Message Body"
13        self.present(composer, animated: true, completion: nil)
14    }
15}
